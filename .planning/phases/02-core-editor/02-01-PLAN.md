**组件设计：**

```vue
<template>
  <div ref="containerRef" class="monaco-editor-container" />
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted, watch } from 'vue'
import loader from '@monaco-editor/loader'

const props = defineProps<{
  modelValue: string
  readonly?: boolean
}>()

const emit = defineEmits<{
  'update:modelValue': [value: string]
  'change': [value: string]
}>()

const containerRef = ref<HTMLElement | null>(null)
let editor: any = null
let monacoInstance: any = null

onMounted(async () => {
  // 1. 动态加载 Monaco Editor（避免 SSR 问题）
  monacoInstance = await loader.init()

  // 2. 注册 Markdown 语法高亮（使用内置 markdown 语言支持）
  // Monaco 内置 markdown 语言，无需额外 monarch tokenizer

  // 3. 配置 Markdown 自动补全（基础）
  monacoInstance.languages.registerCompletionItemProvider('markdown', {
    provideCompletionItems: (model: any, position: any) => {
      const suggestions = [
        // 标题片段
        { label: '# ', kind: monacoInstance.languages.CompletionItemKind.Snippet,
          insertText: '# ${1:Heading}', insertTextRules: monacoInstance.languages.CompletionItemInsertTextRule.InsertAsSnippet },
        // 列表片段
        { label: '- ', kind: monacoInstance.languages.CompletionItemKind.Snippet,
          insertText: '- ${1:Item}', insertTextRules: monacoInstance.languages.CompletionItemInsertTextRule.InsertAsSnippet },
        // 粗体
        { label: '** ', kind: monacoInstance.languages.CompletionItemKind.Snippet,
          insertText: '**${1:text}**', insertTextRules: monacoInstance.languages.CompletionItemInsertTextRule.InsertAsSnippet },
      ]
      return { suggestions }
    }
  })

  // 4. 创建编辑器实例
  editor = monacoInstance.editor.create(containerRef.value!, {
    value: props.modelValue || '',
    language: 'markdown',
    theme: 'vs-dark',  // per D-02
    fontSize: 14,      // per D-03
    lineHeight: 22.4,  // per D-03: 14 * 1.6 = 22.4
    wordWrap: 'on',
    minimap: { enabled: false },
    scrollBeyondLastLine: false,
    automaticLayout: true,  // 响应容器大小变化
    // 编辑功能 per D-05
    multiCursorModifier: 'ctrlCmd',
    enableSmartSelect: true,
    folding: true,
    foldingStrategy: 'indentation',
    showFoldingControls: 'mouseover',
    renderLineHighlight: 'line',
    cursorBlinking: 'smooth',
    smoothScrolling: true,
    // Markdown 特定配置
    lineNumbers: 'on',
    glyphMargin: true,  // 行级交互需要 glyph margin
    renderWhitespace: 'none',
    links: true,
    // 键盘快捷键
    accessibilitySupport: 'auto',
    // 性能优化
    fastScrollSensitivity: 5,
    mouseWheelScrollSensitivity: 1,
  })

  // 5. 内容变化监听
  editor.onDidChangeModelContent(() => {
    const content = editor.getValue()
    emit('update:modelValue', content)
    emit('change', content)
  })

  // 6. 中文输入法支持 per D-04
  editor.updateOptions({
    // IME 输入法支持已默认启用
  })
})

onUnmounted(() => {
  if (editor) {
    editor.dispose()
    editor = null
  }
})

// 外部传入值变化时同步到编辑器
watch(() => props.modelValue, (newVal) => {
  if (editor && newVal !== editor.getValue()) {
    const position = editor.getPosition()
    editor.setValue(newVal || '')
    if (position) editor.setPosition(position)
  }
})

// 对外暴露编辑器实例（用于行级交互 Plan 02-04）
defineExpose({
  getEditor: () => editor,
  getMonaco: () => monacoInstance,
})
</script>

<style scoped>
.monaco-editor-container {
  width: 100%;
  height: 100%;
  overflow: hidden;
}
</style>
```
