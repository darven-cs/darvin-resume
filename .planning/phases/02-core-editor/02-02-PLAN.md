**组件设计：**

```vue
<template>
  <div
    class="split-pane"
    :class="{ 'is-dragging': isDragging }"
    :style="containerStyle"
  >
    <!-- 左侧面板 -->
    <div
      class="pane left-pane"
      :style="leftPaneStyle"
    >
      <slot name="left" />
    </div>

    <!-- 分割线 -->
    <div
      class="divider"
      :class="{ dragging: isDragging }"
      @mousedown="startDrag"
      @mouseenter="hoverDivider"
      @mouseleave="leaveDivider"
    >
      <div class="divider-handle" />
    </div>

    <!-- 右侧面板 -->
    <div
      class="pane right-pane"
      :style="rightPaneStyle"
    >
      <slot name="right" />
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, onUnmounted } from 'vue'

const props = withDefaults(defineProps<{
  defaultRatio?: number   // 默认 50 per D-06
  minWidth?: number        // 每栏最小宽度 300px per D-07
}>(), {
  defaultRatio: 50,
  minWidth: 300,
})

const splitRatio = ref(props.defaultRatio)
const isDragging = ref(false)

const containerStyle = computed(() => ({
  display: 'flex',
  height: '100%',
  width: '100%',
  userSelect: isDragging.value ? 'none' : 'auto',
}))

const leftPaneStyle = computed(() => ({
  width: `${splitRatio.value}%`,
  minWidth: `${props.minWidth}px`,
  overflow: 'hidden',
}))

const rightPaneStyle = computed(() => ({
  width: `${100 - splitRatio.value}%`,
  minWidth: `${props.minWidth}px`,
  overflow: 'hidden',
}))

function startDrag(e: MouseEvent) {
  isDragging.value = true
  const startX = e.clientX
  const startRatio = splitRatio.value

  const onMove = (e: MouseEvent) => {
    const container = document.querySelector('.split-pane') as HTMLElement
    if (!container) return
    const containerWidth = container.offsetWidth
    const delta = e.clientX - startX
    const deltaRatio = (delta / containerWidth) * 100
    const newRatio = Math.max(
      (props.minWidth / containerWidth) * 100,
      Math.min(
        100 - (props.minWidth / containerWidth) * 100,
        startRatio + deltaRatio
      )
    )
    splitRatio.value = Math.max(20, Math.min(80, newRatio))  // per D-07: 限制范围
  }

  const onUp = () => {
    isDragging.value = false
    document.removeEventListener('mousemove', onMove)
    document.removeEventListener('mouseup', onUp)
    document.body.style.cursor = ''
    document.body.style.userSelect = ''
  }

  document.addEventListener('mousemove', onMove)
  document.addEventListener('mouseup', onUp)
}

function hoverDivider() {
  if (!isDragging.value) {
    document.body.style.cursor = 'col-resize'
  }
}

function leaveDivider() {
  if (!isDragging.value) {
    document.body.style.cursor = ''
  }
}

onUnmounted(() => {
  document.body.style.cursor = ''
  document.body.style.userSelect = ''
})
</script>
```

### 2a. A4Page.vue — A4 预览容器组件

```vue
<template>
  <div class="preview-container" ref="containerRef">
    <div
      v-for="(pageContent, index) in pages"
      :key="index"
      class="a4-page"
      :data-page="index + 1"
    >
      <div class="page-content" v-html="pageContent" />
      <div class="page-number">{{ index + 1 }}</div>
    </div>
    <div v-if="pages.length === 0" class="preview-empty">
      开始编写 Markdown，预览将实时显示在这里
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, watch } from 'vue'
import { renderMarkdown } from '../utils/markdown'

const props = defineProps<{
  content: string
}>()

const containerRef = ref<HTMLElement | null>(null)

// A4 尺寸（mm转px, 96dpi: 1mm = 3.78px）
// A4: 210mm × 297mm → 794px × 1123px（近似值）
// 预览区按比例缩放以适应容器宽度
const A4_WIDTH_MM = 210
const A4_HEIGHT_MM = 297
const MM_TO_PX = 3.78

const pageHeightPx = A4_HEIGHT_MM * MM_TO_PX

// 按 A4 高度分页
const pages = computed(() => {
  if (!props.content) return []
  const html = renderMarkdown(props.content)

  // 简单策略：按内容估算分页
  // 实际分页逻辑在后续 Phase 优化（Phase 5 PDF 导出时完善）
  // 这里先实现单页无限滚动模式，分页线作为视觉参考
  return [html]
})
</script>
```
