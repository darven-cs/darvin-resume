```go
package backup

import (
    "context"
    "log"
    "sync"
    "time"

    "Darvin-Resume/internal/crypto"
)

// Scheduler 定时备份调度器
type Scheduler struct {
    mu       sync.Mutex
    interval time.Duration
    cancel   context.CancelFunc
    wg       sync.WaitGroup
    running  bool
}

// NewScheduler 创建备份调度器
func NewScheduler(interval time.Duration) *Scheduler {
    if interval < 5*time.Minute {
        interval = 5 * time.Minute
    }
    return &Scheduler{
        interval: interval,
    }
}

// Start 启动定时备份
func (s *Scheduler) Start(ctx context.Context) {
    s.mu.Lock()
    defer s.mu.Unlock()

    if s.running {
        return
    }

    schedCtx, cancel := context.WithCancel(ctx)
    s.cancel = cancel
    s.running = true

    s.wg.Add(1)
    go s.run(schedCtx)
}

// run 定时任务主循环
func (s *Scheduler) run(ctx context.Context) {
    defer s.wg.Done()

    ticker := time.NewTicker(s.interval)
    defer ticker.Stop()

    for {
        select {
        case <-ctx.Done():
            s.mu.Lock()
            s.running = false
            s.mu.Unlock()
            return

        case <-ticker.C:
            s.performBackup()
        }
    }
}

// performBackup 执行自动备份
func (s *Scheduler) performBackup() {
    deviceKey, err := crypto.DeviceKey()
    if err != nil {
        log.Printf("[BackupScheduler] get device key failed: %v", err)
        return
    }

    path, err := CreateBackup(deviceKey)
    if err != nil {
        log.Printf("[BackupScheduler] backup failed: %v", err)
        return
    }
    log.Printf("[BackupScheduler] auto backup created: %s", path)
}

// Stop 停止定时备份
func (s *Scheduler) Stop() {
    s.mu.Lock()
    defer s.mu.Unlock()

    if !s.running {
        return
    }

    if s.cancel != nil {
        s.cancel()
    }
    s.mu.Unlock()  // 先解锁，让 run goroutine 能获取锁
    s.wg.Wait()
    s.mu.Lock()    // 重新获取锁
    s.running = false
}

// IsRunning 返回调度器是否在运行
func (s *Scheduler) IsRunning() bool {
    s.mu.Lock()
    defer s.mu.Unlock()
    return s.running
}

// SetInterval 更新备份间隔（需要重启调度器才能生效）
func (s *Scheduler) SetInterval(interval time.Duration) {
    s.mu.Lock()
    defer s.mu.Unlock()
    s.interval = interval
}
```

**关键设计决策：**
- 使用 `context.WithCancel` 实现优雅关闭
- 使用 `sync.WaitGroup` 确保备份完成后才退出
- 最小间隔 5 分钟，防止过于频繁
- 自动备份使用设备密钥加密，用户无需输入密码
- Stop() 中先 cancel context，再 WaitGroup.Wait()

func ParseInterval(interval string) time.Duration {
    switch interval {
    case "daily":
        return 24 * time.Hour
    case "weekly":
        return 7 * 24 * time.Hour
    case "monthly":
        return 30 * 24 * time.Hour
    default:
        return 24 * time.Hour // 默认每日
    }
}

// GetAutoBackupSettings 获取自动备份设置
func (a *App) GetAutoBackupSettings() (string, error) {
    enabled, _ := settings.Get(a.ctx, "backup.autoEnabled")
    interval, _ := settings.Get(a.ctx, "backup.autoInterval")
    result := map[string]string{
        "enabled":  enabled,
        "interval": interval,
    }
    data, _ := json.Marshal(result)
    return string(data), nil
}

// SetAutoBackupSettings 设置自动备份
func (a *App) SetAutoBackupSettings(enabled bool, interval string) error {
    enabledStr := "false"
    if enabled {
        enabledStr = "true"
    }
    if err := settings.Set(a.ctx, "backup.autoEnabled", enabledStr); err != nil {
        return err
    }
    if err := settings.Set(a.ctx, "backup.autoInterval", interval); err != nil {
        return err
    }

    // 更新调度器
    if a.backupSched != nil {
        a.backupSched.Stop()
        if enabled {
            dur := backup.ParseInterval(interval)
            a.backupSched = backup.NewScheduler(dur)
            a.backupSched.Start(a.ctx)
        }
    }
    return nil
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| 定时调度器 → 文件系统 | 定时备份写入磁盘，需要确保磁盘空间足够 |
| 设备密钥 → 自动备份 | 自动备份使用设备密钥加密，重装系统后无法恢复自动备份 |

## Mitigations
- 最大保留 10 个备份文件，防止磁盘空间耗尽
- 自动备份使用设备密钥加密，手动导出备份可用用户密码
- 调度器可随时禁用
</threat_model>
