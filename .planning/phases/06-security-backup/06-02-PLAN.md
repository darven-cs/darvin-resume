```go
import (
    "Darvin-Resume/internal/backup"
)

// CreateManualBackup 创建手动备份（用户可选密码）
// password 为空表示不加密
func (a *App) CreateManualBackup(password string) (string, error) {
    path, err := backup.CreateBackup(password)
    if err != nil {
        return "", err
    }
    return path, nil
}

// RestoreFromBackup 从备份文件恢复
func (a *App) RestoreFromBackup(filePath string, password string) error {
    return backup.RestoreBackup(filePath, password)
}

// ListBackups 列出所有本地备份
func (a *App) ListBackups() (string, error) {
    // 返回 JSON 字符串（前端解析）
    backups, err := backup.ListBackups()
    if err != nil {
        return "", err
    }
    data, err := json.Marshal(backups)
    if err != nil {
        return "", err
    }
    return string(data), nil
}

// ShowSaveBackupDialog 打开文件保存对话框选择备份保存路径
func (a *App) ShowSaveBackupDialog() (string, error) {
    path, err := runtime.SaveFileDialog(a.ctx, runtime.SaveDialogOptions{
        Title: "导出备份",
        DefaultFilename: fmt.Sprintf("darvin-resume-backup-%s.darvin-backup",
            time.Now().Format("20060102-150405")),
        Filters: []runtime.FileFilter{
            {DisplayName: "Darvin Resume 备份", Pattern: "*.darvin-backup"},
        },
    })
    return path, err
}

// ShowOpenBackupDialog 打开文件选择对话框选择备份文件
func (a *App) ShowOpenBackupDialog() (string, error) {
    path, err := runtime.OpenFileDialog(a.ctx, runtime.OpenDialogOptions{
        Title: "选择备份文件",
        Filters: []runtime.FileFilter{
            {DisplayName: "Darvin Resume 备份", Pattern: "*.darvin-backup"},
        },
    })
    return path, err
}

// ExportBackupToPath 导出备份到指定路径
func (a *App) ExportBackupToPath(savePath string, password string) (string, error) {
    // 创建备份到临时位置，然后复制到用户指定路径
    tmpPath, err := backup.CreateBackup(password)
    if err != nil {
        return "", err
    }
    // 复制文件到 savePath
    data, err := os.ReadFile(tmpPath)
    if err != nil {
        return "", err
    }
    if err := os.WriteFile(savePath, data, 0644); err != nil {
        return "", err
    }
    return savePath, nil
}
```

```typescript
import { ref, reactive } from 'vue'

export interface BackupInfo {
  filename: string
  path: string
  createdAt: string
  encrypted: boolean
  dbSize: number
  tables: Record<string, { count: number }>
}

export function useBackup() {
  const isExporting = ref(false)
  const isImporting = ref(false)
  const backupList = ref<BackupInfo[]>([])
  const error = ref<string>('')

  // 导出备份
  async function exportBackup(password: string = '') {
    isExporting.value = true
    error.value = ''
    try {
      // 1. 选择保存路径
      const savePath = await window.go.main.App.ShowSaveBackupDialog()
      if (!savePath) return null // 用户取消

      // 2. 创建并保存备份
      const result = await window.go.main.App.ExportBackupToPath(savePath, password)
      return result
    } catch (e: any) {
      error.value = e.message || '导出失败'
      throw e
    } finally {
      isExporting.value = false
    }
  }

  // 导入恢复
  async function importBackup(password: string = '') {
    isImporting.value = true
    error.value = ''
    try {
      // 1. 选择备份文件
      const filePath = await window.go.main.App.ShowOpenBackupDialog()
      if (!filePath) return // 用户取消

      // 2. 恢复备份
      await window.go.main.App.RestoreFromBackup(filePath, password)

      // 3. 恢复成功后刷新页面
      window.location.reload()
    } catch (e: any) {
      error.value = e.message || '恢复失败'
      throw e
    } finally {
      isImporting.value = false
    }
  }

  // 加载备份列表
  async function loadBackupList() {
    try {
      const json = await window.go.main.App.ListBackups()
      if (json) {
        backupList.value = JSON.parse(json)
      }
    } catch (e: any) {
      error.value = e.message || '加载备份列表失败'
    }
  }

  return {
    isExporting,
    isImporting,
    backupList,
    error,
    exportBackup,
    importBackup,
    loadBackupList,
  }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| 备份文件 → 文件系统 | 备份文件可能包含敏感数据（API Key、简历内容），加密保护 |
| VACUUM INTO 临时文件 | 临时文件包含完整数据库快照，需要安全清理 |
| 恢复操作 | 恢复会替换当前所有数据，需要 pre-restore 备份保护 |

## Mitigations
- 可选密码加密（AES-256-GCM）
- 临时文件用后立即删除（defer os.Remove）
- 恢复前自动创建 pre-restore 备份
- 恢复失败时回滚到 pre-restore
- 校验 backup.json 的 App 标识防止错误文件导入
- SHA-256 校验和验证数据完整性
</threat_model>
