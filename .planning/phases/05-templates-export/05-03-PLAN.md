async function exportWithSystemPrint() {
  // 1. 应用分页线显示/隐藏
  document.body.classList.toggle('hide-page-breaks', !options.hidePageBreaks)

  // 2. 将预览区包装到 .print-container 中

}

```go
package export

import (
	"context"
	"fmt"
	"net/url"
	"os"

	"github.com/chromedp/chromedp"
	"github.com/chromedp/chromedp/cdproto/page"
)

// PDFOptions PDF导出配置
type PDFOptions struct {
	PaperWidth  float64 // inches
	PaperHeight float64 // inches
	Scale       float64 // DPI / 72
	PrintBg     bool    // 打印背景色
}

// DefaultPDFOptions 返回默认 A4 PDF 选项
func DefaultPDFOptions() *PDFOptions {
	return &PDFOptions{
		PaperWidth:  8.27,  // A4: 210mm = 8.27 inches
		PaperHeight: 11.69, // A4: 297mm = 11.69 inches
		Scale:       1.0,  // DPI 96 / 72
		PrintBg:     true,
	}
}

// ExportPDFFromHTML 使用 Chromedp 从 HTML 内容导出 PDF
func ExportPDFFromHTML(ctx context.Context, htmlContent string, outputPath string, opts *PDFOptions) error {
	if opts == nil {
		opts = DefaultPDFOptions()
	}

	// 创建无头浏览器分配器
	allocCtx, cancel := chromedp.NewExecAllocator(ctx,
		chromedp.Flag("headless", true),
		chromedp.Flag("disable-gpu", true),
		chromedp.Flag("no-sandbox", true),
		chromedp.Flag("disable-setuid-sandbox", true),
	)
	defer cancel()

	// 创建浏览器上下文
	browserCtx, cancel2 := chromedp.NewContext(allocCtx)
	defer cancel2()

	var pdfBuf []byte
	err := chromedp.Run(browserCtx,
		// 加载 HTML 内容（使用 data: URL）
		chromedp.Navigate("data:text/html,"+url.QueryEscape(htmlContent)),
		chromedp.WaitReady("body"),

		// 生成 PDF
		chromedp.ActionFunc(func(ctx context.Context) error {
			p := page.PrintToPDF().
				WithPrintBackground(opts.PrintBg).
				WithLandscape(false).
				WithPaperWidth(opts.PaperWidth).
				WithPaperHeight(opts.PaperHeight).
				WithMarginTop(0).
				WithMarginBottom(0).
				WithMarginLeft(0).
				WithMarginRight(0).
				WithScale(opts.Scale).
				WithPrintSelectionOnly(false).
				WithPreferCSSPageSize(true)

			buf, _, err := p.Do(ctx)
			if err != nil {
				return fmt.Errorf("PrintToPDF failed: %w", err)
			}
			pdfBuf = buf
			return nil
		}),
	)

	if err != nil {
		return fmt.Errorf("chromedp run failed: %w", err)
	}

	// 写入文件
	if err := os.WriteFile(outputPath, pdfBuf, 0644); err != nil {
		return fmt.Errorf("write file failed: %w", err)
	}

	return nil
}
```

defineExpose({

function getHTMLForExport(): string {
  // 构建完整的打印用 HTML 字符串
  // 包含：<!DOCTYPE html> + html/head/style + body + .a4-page 页面内容
  const pagesHtml = pages.value.map((content, index) => {
    return `<div class="a4-page template-${props.templateId || 'minimal'}" data-page="${index + 1}">
      <div class="page-content">${content}</div>
      <div class="page-number">${index + 1}</div>
    </div>`
  }).join('\n')

  // 注入 editor.css 内容
  const cssContent = getComputedStyle(document.documentElement)
  // 从静态 import 中获取 CSS 内容（通过 fetch 或内联）
  // 简化：直接使用 CSS 字符串
  const templateCSS = loadTemplateCSS(props.templateId || 'minimal')
  const customCSS = props.customCss || ''
  const varsCSS = buildCSSVars(getCurrentVars())

  return `<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<style>
/* 基础样式 */
${getBaseCSSRules()}
/* CSS 变量 */
${varsCSS}
/* 模板样式 */
${templateCSS}
/* 自定义样式 */
${customCSS}
/* 打印样式 */
${getPrintRules()}
/* 分页线隐藏 */
${options?.hidePageBreaks ? '' : ''}
</style>
</head>
<body>
<div class="print-container">
${pagesHtml}
</div>
</body>
</html>`
}

// 辅助函数
function loadTemplateCSS(templateId: string): string {
  // 返回对应模板的 CSS 字符串
  // 可以通过 import.meta.glob 动态加载
  const cssMap: Record<string, string> = {}
  // 返回内联 CSS 字符串
  return cssMap[templateId] || ''
}

function getPrintRules(): string {
  // 返回 @media print 规则
  return `
@media print {
  .a4-page { width: 210mm; height: auto; padding: 20mm; margin: 0; box-shadow: none; page-break-after: always; }
  .page-content > * { break-inside: avoid; -webkit-column-break-inside: avoid; }
  * { -webkit-print-color-adjust: exact; print-color-adjust: exact; }
  .a4-page::before, .a4-page::after { display: none; }
}`
}

function getHTMLForExport(): string {
  // 从 DOM 中获取渲染后的完整 HTML
  const container = containerRef.value

}

| backend -> filesystem | PDF 写入用户指定的 outputPath，需验证路径合法性 |
