```typescript
import { ref, computed, watch } from 'vue'

// 模板定义
export interface TemplateDef {
  id: string       // 'minimal' | 'dual-col' | 'academic' | 'campus'
  name: string     // 显示名称
  nameEn: string   // 英文名
  category: 'builtin' | 'user'
}

export const BUILTIN_TEMPLATES: TemplateDef[] = [
  { id: 'minimal', name: '极简通用', nameEn: 'Minimal', category: 'builtin' },
  { id: 'dual-col', name: '双栏简约', nameEn: 'Dual Column', category: 'builtin' },
  { id: 'academic', name: '学术科研', nameEn: 'Academic', category: 'builtin' },
  { id: 'campus', name: '大厂校招', nameEn: 'Campus', category: 'builtin' },
]

export function useTemplate(resumeId: Ref<string>) {
  const currentTemplateId = ref<string>('minimal')
  const customCss = ref<string>('')
  const isLoading = ref(false)

  // 加载简历当前模板设置（从后端获取）
  async function loadTemplate() {
    isLoading.value = true
    try {
      // 调用后端获取简历的 templateId 和 customCss
      const resume = await window.go.main.App.GetResume(resumeId.value)
      currentTemplateId.value = resume.templateId || 'minimal'
      customCss.value = resume.customCss || ''
    } finally {
      isLoading.value = false
    }
  }

  // 选择模板（更新 templateId 并保存到后端）
  async function selectTemplate(templateId: string) {
    if (templateId === currentTemplateId.value) return
    currentTemplateId.value = templateId
    await window.go.main.App.UpdateResumeTemplate(resumeId.value, templateId)
  }

  // 保存为个人模板（更新 customCss）
  async function saveAsTemplate(css: string) {
    customCss.value = css
    await window.go.main.App.UpdateResumeCustomCSS(resumeId.value, css)
  }

  // 获取当前模板 CSS 内容
  const currentTemplateCSS = computed(() => {
    const cssMap: Record<string, string> = {
      minimal: 'template-minimal.css',
      'dual-col': 'template-dual-col.css',
      academic: 'template-academic.css',
      campus: 'template-campus.css',
    }
    return cssMap[currentTemplateId.value] || 'template-minimal.css'
  })

  return {
    currentTemplateId,
    customCss,
    isLoading,
    loadTemplate,
    selectTemplate,
    saveAsTemplate,
    currentTemplateCSS,
    templates: BUILTIN_TEMPLATES,
  }
}
```

注意：useTemplate 需要在 EditorView 中被调用，接收当前 resumeId。

```typescript
import { watch, onMounted, onUnmounted } from 'vue'
import { renderMarkdown } from '../utils/markdown'
import '../styles/editor.css'
// 导入所有模板 CSS（静态导入，确保打包）
import '../styles/templates/template-minimal.css'
import '../styles/templates/template-dual-col.css'
import '../styles/templates/template-academic.css'
import '../styles/templates/template-campus.css'

// 动态注入自定义 CSS
const dynamicStyleId = 'template-dynamic-css'

watch([() => props.templateId, () => props.customCss], ([tid, css]) => {
  // 移除旧标签
  const old = document.getElementById(dynamicStyleId)
  if (old) old.remove()

  if (css) {
    const style = document.createElement('style')
    style.id = dynamicStyleId
    style.textContent = css
    // 注入到 A4Page 容器中，作用域限定在 .a4-page 内
    containerRef.value?.appendChild(style)
  }
}, { immediate: true })
```

3. **模板类名注入：**
在 `.a4-page` 容器上添加动态 class：`class="a4-page" :class="template-class"`，其中 template-class 根据 `templateId` 生成：
- minimal -> `template-minimal`
- dual-col -> `template-dual-col`
- academic -> `template-academic`
- campus -> `template-campus`

4. **清理：** 在 `onUnmounted` 中移除动态 style 标签

```go
// UpdateTemplateID 更新简历模板 ID
func (s *resumeService) UpdateTemplateID(ctx context.Context, id string, templateId string) error {
    // 验证 templateId 为有效值
    validIds := map[string]bool{"minimal": true, "dual-col": true, "academic": true, "campus": true}
    if !validIds[templateId] {
        return errors.New("invalid template id")
    }
    query := `UPDATE resumes SET template_id = ?, updated_at = ? WHERE id = ? AND is_deleted = FALSE`
    now := time.Now()
    result, err := database.DB.ExecContext(ctx, query, templateId, now, id)
    if err != nil {
        return err
    }
    rows, err := result.RowsAffected()
    if err != nil {
        return err
    }
    if rows == 0 {
        return ErrResumeNotFound
    }
    return nil
}

// UpdateCustomCSS 更新简历自定义 CSS
func (s *resumeService) UpdateCustomCSS(ctx context.Context, id string, customCss string) error {
    query := `UPDATE resumes SET custom_css = ?, updated_at = ? WHERE id = ? AND is_deleted = FALSE`
    now := time.Now()
    result, err := database.DB.ExecContext(ctx, query, customCss, now, id)
    if err != nil {
        return err
    }
    rows, err := result.RowsAffected()
    if err != nil {
        return err
    }
    if rows == 0 {
        return ErrResumeNotFound
    }
    return nil
}
```
